Swift - Chigoo Chat App Message System 
// Message Model mit Codable für Firebase
struct Message: Codable, Identifiable {
    let id: String
    let text: String
    let senderId: String
    let timestamp: Date
    var isRead: Bool = false
}

// Chat Service mit async/await
class ChatService: ObservableObject {
    @Published var messages: [Message] = []
    private let db = Firestore.firestore()
    
    func sendMessage(_ text: String, to userId: String) async throws {
        let message = Message(
            id: UUID().uuidString,
            text: text,
            senderId: Auth.auth().currentUser?.uid ?? "",
            timestamp: Date()
        )
        
        try await db.collection("chats")
            .document(userId)
            .collection("messages")
            .addDocument(data: message.asDictionary())
    }
}
Kotlin - Folk Music App MVVM Architecture 
// ViewModel mit Kotlin Coroutines
class MusicViewModel(
    private val repository: MusicRepository
) : ViewModel() {
    
    private val _musicList = MutableLiveData<List<Song>>()
    val musicList: LiveData<List<Song>> = _musicList
    
    private val _isPlaying = MutableStateFlow(false)
    val isPlaying = _isPlaying.asStateFlow()
    
    fun loadMusic() {
        viewModelScope.launch {
            try {
                val songs = repository.getAllSongs()
                _musicList.value = songs
            } catch (e: Exception) {
                Log.e("MusicViewModel", "Error: ${e.message}")
            }
        }
    }
    
    fun playSong(song: Song) {
        viewModelScope.launch(Dispatchers.IO) {
            repository.playSong(song)
            _isPlaying.emit(true)
        }
    }
}

Firebase Authentication Service 
// Firebase Auth Service mit Error Handling
class AuthenticationService: ObservableObject {
    @Published var user: User?
    @Published var isAuthenticated = false
    
    init() {
        setupAuthListener()
    }
    
    private func setupAuthListener() {
        Auth.auth().addStateDidChangeListener { _, user in
            self.user = user
            self.isAuthenticated = user != nil
        }
    }
    
    func signIn(email: String, password: String) async throws -> AuthDataResult {
        do {
            let result = try await Auth.auth().signIn(
                withEmail: email,
                password: password
            )
            
            // Update user profile in Firestore
            try await updateUserProfile(userId: result.user.uid)
            
            return result
        } catch {
            throw AuthError.signInFailed(error.localizedDescription)
        }
    }
}
